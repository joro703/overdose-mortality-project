# backward elimination using 10-fold cross validation
k = 10
set.seed(1)

folds = sample(1:k, nrow(data_hs), replace = TRUE)

cv.errors = matrix(NA, k, 3, dimnames = list(NULL, paste(1:3)))

# function for prediction with regsubsets
predict.regsubsets = function(object, newdata, id, ...){
  form = as.formula(object$call[[2]])
  mat = model.matrix(form, newdata)
  coefi = coef(object, id = id)
  xvars = names(coefi)
  mat[, xvars] %*% coefi
}

for(j in 1:k){
  best.fit = regsubsets(alldrug_rate ~ med_income + poverty + hs_grad, 
                        data = data_hs[folds != j, ],
                        nvmax = 3,
                        method = "backward")
  
  for(i in 1:3){
    pred = predict(best.fit, data_hs[folds == j, ], id = i)
    cv.errors[j, i] = mean((data_hs$alldrug_rate[folds == j] - pred)^2)
  }
}

mean.cv.errors = apply(cv.errors, 2, mean)

plot(mean.cv.errors, type = "b")

# perform best subset selection and obtain the model
reg.best_hs = regsubsets(alldrug_rate ~ med_income + poverty + hs_grad, 
                         data = data_hs, nvmax = 3, method = "backward")
coef(reg.best_hs, which.min(mean.cv.errors))


# obtained model
alldrug_model_hs = lm(alldrug_rate ~ med_income + poverty + hs_grad,

cv.mae = rep(NA, k)

# best lambda

# final ridge model
ridge.mod = glmnet(x, y, alpha = 0, lambda = bestlam.ridge)

# coefficients

# predictions using 10-fold CV-selected lambda
ridge.pred = predict(cv.out.ridge, s = "lambda.min", newx = x)

actual = y

                 data = data_heroin)[, -1]

y = data_heroin$heroin_rate

set.seed(1)

# 10-fold cross-validation ridge regression
cv.out.ridge = cv.glmnet(x, y, alpha = 0, nfolds = 10)

# best lambda
bestlam.ridge = cv.out.ridge$lambda.min

# final ridge model
ridge.mod = glmnet(x, y, alpha = 0, lambda = bestlam.ridge)

# coefficients
coef(ridge.mod)

# predictions using 10-fold CV-selected lambda
ridge.pred = predict(cv.out.ridge, s = "lambda.min", newx = x)

actual = y

heroin_int.mse = mean((actual - ridge.pred)^2)
heroin_int.rmse = sqrt(heroin_int.mse)
heroin_int.mae = mean(abs(actual - ridge.pred))

# R-squared
heroin_int.r2 = 1 - sum((actual - ridge.pred)^2) /
  sum((actual - mean(actual))^2)

# Adjusted R-squared
n = length(actual)
p = ncol(x)

heroin_int.adjr2 = 1 - ((1 - heroin_int.r2) * (n - 1)) /
  (n - p - 1)
